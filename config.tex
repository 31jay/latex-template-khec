% =========================
% --- Encoding & Fonts ---
% =========================
\usepackage[utf8]{inputenc}       % UTF-8 encoding for input
\usepackage[T1]{fontenc}          % Better font encoding for accented characters
\usepackage{times}                 % Times font
\usepackage{newtxtext,newtxmath}  % Times-like text and math fonts

% =========================
% --- Page Layout ---
% =========================
\usepackage[top=1in,bottom=1in,left=1.5in,right=1in]{geometry}  % Page margins
\setcounter{secnumdepth}{3}  % Number sections down to subsubsection
\setcounter{tocdepth}{3}     % Include sections down to subsubsection in ToC

% =========================
% --- Mathematics ---
% =========================
\usepackage{amsmath,amssymb,amsfonts,bbm}  % Math symbols and fonts

% =========================
% --- Graphics ---
% =========================
\usepackage{graphicx}        % Include images
\usepackage{subcaption}      % Subfigures support
\usepackage{float}           % Force figure/table placement with [H]

% =========================
% --- Bibliography ---
% =========================
\usepackage[numbers]{natbib}  % Numeric citations

% =========================
% --- Hyperlinks ---
% =========================
\usepackage[hidelinks]{hyperref}  % Clickable links without color boxes
\usepackage{url}                   % For proper formatting of URLs

% =========================
% --- Lists ---
% =========================
\usepackage{enumitem}  % Customize lists

% =========================
% --- Typography & Spacing ---
% =========================
\usepackage{microtype}   % Better typography (kerning, spacing)
\usepackage{parskip}     % Adds space between paragraphs
\newcommand\tab[1][1cm]{\hspace*{#1}}  % Custom horizontal spacing

% =========================
% --- Hyphenation & Line Breaking ---
% =========================
\usepackage[none]{hyphenat}  % Disable automatic hyphenation
\sloppy                       % Looser line breaking to avoid overflow
\setlength{\emergencystretch}{1.5em}  % Extra stretch to help line breaks

% =========================
% --- Chapter Formatting ---
% =========================
\usepackage[pagestyles]{titlesec}

% Custom chapter style
\titleformat
{\chapter}          % Command being redefined
[display]            % Display style (chapter title on its own line)
{\Huge \bfseries}    % Format of chapter title
{Chapter \thechapter} % Label format (e.g., "Chapter 1")
{1ex}                % Separation between label and title text
{}                   % Before title code
[
    \vspace{0ex}    % After title code (extra spacing)
]

% Chapter spacing: {left}{before}{after}
\titlespacing*{\chapter}{0ex}{-2ex}{4ex}

% =========================
% --- Symbols ---
% =========================
\usepackage{pifont}  % Dingbats symbols (checkmarks, crosses, etc.)

% For abbreviations
\usepackage{nomencl}
\makenomenclature


% for table formatting
\usepackage{tabularx}
\usepackage{xcolor, colortbl} % For cell colors
\usepackage{diagbox}
\usepackage{multirow}
